S3 · CloudFront · Lambda@Edge · Terraform

infra.tf 
# Despliegue automático vía OIDC + GitHub Actions
module "cdn" {
  source  = "./modules/cloudfront"

  origin          = "s3-assets-prod"
  oac             = true
  tls             = "ACM (us-east-1)"
  edge_function   = aws_lambda_function.security_headers
  price_class     = "PriceClass_100"
}

resource "aws_s3_bucket_lifecycle" "uploads" {
  transition_to_ia       = 30   # días
  transition_to_glacier  = 90
  expiration             = 180
}

console.log("→ https://d14x2vaw9f2mo1.cloudfront.net")
